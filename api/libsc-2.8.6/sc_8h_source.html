365/* hopefully fast binary logarithms and binary round up */
366
367#define SC_LOG2_8(x) (sc_log2_lookup_table[(x)])
368#define SC_LOG2_16(x) (((x) > 0xff) ? \
369 (SC_LOG2_8 ((x) >> 8) + 8) : SC_LOG2_8 (x))
370#define SC_LOG2_32(x) (((x) > 0xffff) ? \
371 (SC_LOG2_16 ((x) >> 16)) + 16 : SC_LOG2_16 (x))
372#define SC_LOG2_64(x) (((x) > 0xffffffffLL) ? \
373 (SC_LOG2_32 ((x) >> 32)) + 32 : SC_LOG2_32 (x))
374#define SC_ROUNDUP2_32(x) \
375 (((x) <= 0) ? 0 : (1 << (SC_LOG2_32 ((x) - 1) + 1)))
376#define SC_ROUNDUP2_64(x) \
377 (((x) <= 0) ? 0 : (1LL << (SC_LOG2_64 ((x) - 1LL) + 1)))
